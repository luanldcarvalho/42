#include "libft.h"
size_t	ft_strlcpy(char *dst, const char *src, size_t size) // Copia na dst n bytes da src e retorna o tamanho da string que foi copiada, nesse caso o tamanho inicial de src
{
	size_t	i; // Index em size_t para comparar com o size_t de size

	if (!dst || !src) // Se dst ou src nao existir, ou seja for nulo
		return (0); // Retorna 0
	i = 0;
	if (size == 0) // Se o size passado for 0, retorna o tamanho inicial de src
		return (ft_strlen(src));
	while (i < size - 1 && src[i]) // Enquanto o index for menor que tamanho - 1 (os index começam em 0 nao em 1) e a src no index existir
	{
		dst[i] = src[i]; // dst no index recebe o valor de src no index
		i++; // incrementa o index
	}
	dst[i] = '\0'; // Por fim, quando sai do loop, o ultimo valor de dst no index recebe nulo
	return (ft_strlen(src)); // A função retorna o tamanho de src
}
