#include "libft.h"

int	ft_memcmp(const void *s1, const void *s2, size_t n) // Compara os primeiros n bytes da string1 e da string2.
{
	unsigned char	*us1; // Unsigned char pois irá comparar caractere por caractere
	unsigned char	*us2;

	us1 = (unsigned char *)s1; // Atribuindo string1 em uma variavel us1(Unsigned String1) com o valor de s1 apos passar pelo casting
	us2 = (unsigned char *)s2;
	while (n > 0) // Enquanto n for maior que zero, estaremos comparando os valores de string1 e strig2
	{
		if (*us1 > *us2) // Se caractere de string1 for maior que o caractere de string1 (valores da tabela ASCII) retornamos (1)
			return (1);
		else if (*us1 < *us2) // Se for menor, retornamos (-1)
			return (-1);
		us1++; // Se não é maior nem menor é igual, com isso fazemos o incremento das duas strings para comparar o proximo bytes
		us2++;
		n--; // Mas devemos também decrementar o n, pois temos o limite de n bytes para comparar
	}
	return (0); // Por fim, se nunca encontrar um valor maior ou menor, retornará (0). Com isso sabemos que a string1 e strin2 quando comparado n bytes são iguais.
}
