char	*ft_strrchr(const char *s, int c) // É bastante semelhante a strchr mas, em vez de retornar a primeira ocorrencia do caractere c na string, precisa retornar a ultima ocorrencia. Nesse caso, podemos começar a procurar do fim ao inicio da string
{
	char	*str;
	int		leng_s;

	str = (char *)s; // str aponta para o mesmo lugar que s, recebe os valores apos o casting como char (antes era const char)
	leng_s = ft_strlen(s); // leng_s recebe o tamanho da string
	if (c == '\0') // Se o caractere for nulo, já retorna um ponteiro para o fim da string, que é a string em seu index valendo o tamanho da string
	{
		return (&str[leng_s]);
	}
	leng_s--; // Caso o caractere não seja nulo, já decrementamos 1 do tamanho da string para continuar os calculos
	while (leng_s >= 0) // Enquanto leng_s for maior ou igual a 0, estamos percorrendo pela string
	{
		if (str[leng_s] == c) // Se str no valor de leng_s (que esta recebendo decremento a cada vez) for igual ao char c, é porque encontramos a ultima ocorrencia de c na string
			return (&str[leng_s]); // Retornamos entao um ponteiro para esse local na string s
		leng_s--; // Mas caso o valor de leng_s não seja igual ao caractere, decrementamos o leng_s mais uma vez
	}
	return (NULL); // Se o caractere c não for encontrado nenhuma vez na string, é retornado null
}
