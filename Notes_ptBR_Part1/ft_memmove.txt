#include "libft.h"

void	*ft_memmove(void *dest, const void *src, size_t n) // Copia n bytes de src para uma string, em seguida copiar essa string para destine. É mais lenta que a memcpy mas é mais interessante de usar quando vai sobrepor algum dado na destine.
{
	char		*destine;
	const char	*source; // O src foi passado como parametro const void, por isso aqui tambem e const
	char		*memory;

	memory = malloc(sizeof (n)); // Será alocado em memory n bytes usando a função malloc, como argumento passado sizeof que é tamanha de n
	destine = dest; // destine aponta para o mesmo lugar que dest
	source = src; // source aponta para o mesmo lugar que src
	ft_memcpy(memory, source, n); // copia n bytes de source para memory
	ft_memcpy(destine, memory, n); // copia n bytes de memory para destine
	free (memory); // libera a memoria de memory
	return (dest); // retorna o valor de dest
}
