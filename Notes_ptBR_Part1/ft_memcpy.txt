#include "libft.h"

void	*ft_memcpy(void *dest, const void *src, size_t n) // Copia n bytes da string src para a string dest
{
	unsigned long	index;
	char			*destine;
	const char		*source; // Sem o const temos o erro: assignment discards ‘const’ qualifier from pointer target type. Ou seja, nosso src foi passado com o parametro const void, precisamos usar o const aqui.

	index = 0;
	destine = dest; // destine recebe dest
	source = src; // source recebe src
	while (n > index) // enquanto n que é a quantidade de memoria que iremos copiar for maior que o index continua no loop, assim chegará uma hora em que o index será igual ao n e então estaremos no limite proposto.
	{
		destine[index] = source[index]; // Copiando o char de source em destine
		index++; // Incrementando index
	}
	return (dest); // Retornando dest (O dest também recebeu as mudanças, os dois apontam pro mesmo lugar)
}

// OBS: Você usa constante quando você não quer correr o risco da função alterar alguma coisa inesperada naquele argumento que você passou na função, você quer que ela só leia os dados sem alterar nada. É basicamente uma variável read-only
