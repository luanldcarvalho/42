#include "libft.h"

size_t	ft_strlen(const char *s) // Para descobrir o length da string
{
	int	index;

	index = 0; // Index começa em 0
	while (s[index] != '\0') // Enquanto não está no fim da string
	{
		index ++; // index incrementa
	}
	// Quando a string[index] chega em '\0' sai do loop while
	return (index); // Retorna o index que é o tamanho da string
}
