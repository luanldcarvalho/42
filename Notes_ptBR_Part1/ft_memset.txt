#include "libft.h"

void	*ft_memset(void *s, int c, size_t n) // Preenche uma quantidade de memoria com um determinado valor de byte
{
	unsigned long	index; // unsigned long pois irá comparar com um size_t
	char			*str2;

	index = 0; // index começa em zero
	str2 = s; // str2 recebe a string (s) passada em parametro
	while (index < n) // enquanto index for menor que o n passado em parametro
	{
		str2[index] = (unsigned char)c; // a string2 recebe o char passado. Aqui, o valor int é convertido em char (tabela ascii)
		index ++; // index incrementa
	}
	return (s); // Retorna o s (o s também recebeu as mudanças de str2, porque apontam pro mesmo lugar)
}

// *s a string em que será copiado
// c o caractere que será copiado, o valor é passado em int mas durante a função o mesmo é convertido para unsigned char, pois a tabela ASCII vai de 0 a 255, o unsigned char também
// n será a quantidade copiada
