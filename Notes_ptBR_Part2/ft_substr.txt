#include "libft.h"

char	*ft_substr(char const *s, unsigned int start, size_t len) // Essa função retorna um subtring. Essa substring esta dentro da string passada e tem o tamanho máximo len. O primeiro caractere da substring esta no indice start, ou seja na posição start da string passada
{
	char	*pointer; // Um ponteiro tipo char para apontarmos para a substring
	size_t	index; // Index tipo size_t pois vamos comparar com um unsigned int
	size_t	index2;

	pointer = malloc(len); // pointer recebe uma alocação de memoria de tamanho len, que é o maximo que vamos precisar
	index = 0;
	index2 = 0;
	while (index < start) // Enquanto o index for menor que start, o index incrementa
		index++;
	if (index == start) // Quando o index é igual a start, encontramos onde nossa substring se inicia
	{
		while (len > 0) // Enquanto len é maior que zero, vamos adicionar caracteres na nossa substring
		{
			pointer[index2] = s[index + index2]; // Posição index2 do nosso ponteiro recebe o valor da string no index + index2
			index2++; // Dessa maneira podemos incrementar somente o index2
			len--; // E decrementar o len
		}
		return (pointer); // No fim, temos nossa substring completa e retornamos o ponteiro que aponta para o inicio dessa substring
	}
	return (NULL); // Caso não seja possível retornar a substring, retornamos NULL
}
